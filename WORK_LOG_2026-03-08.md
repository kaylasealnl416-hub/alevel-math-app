### 认证系统实现

**API 客户端增强**（`src/utils/api.js`）：

```javascript
// 获取存储的 token
function getAuthToken() {
  return localStorage.getItem('auth_token')
}

// 设置 token
export function setAuthToken(token) {
  if (token) {
    localStorage.setItem('auth_token', token)
  } else {
    localStorage.removeItem('auth_token')
  }
}

// 统一请求函数（自动添加 token）
async function backendRequest(endpoint, options = {}) {
  const token = getAuthToken()
  const config = {
    headers: {
      'Content-Type': 'application/json',
      ...(token && { 'Authorization': `Bearer ${token}` }),
      ...options.headers,
    },
    ...options,
  }
  // ... 发送请求
}
```

**认证 API**：
```javascript
export const authAPI = {
  // 用户注册
  register: async (userData) => {
    const response = await backendRequest('/api/auth/register', {
      method: 'POST',
      body: JSON.stringify(userData)
    })
    if (response.data?.token) {
      setAuthToken(response.data.token)
    }
    return response.data
  },

  // 用户登录
  login: async (credentials) => {
    const response = await backendRequest('/api/auth/login', {
      method: 'POST',
      body: JSON.stringify(credentials)
    })
    if (response.data?.token) {
      setAuthToken(response.data.token)
    }
    return response.data
  },

  // 登出
  logout: () => {
    setAuthToken(null)
  }
}
```

### 数据库 Schema 修复

**问题**：
- 数据库中的 `users` 表缺少 `password` 列
- 导致注册 API 返回 500 错误：`column "password" does not exist`

**原因**：
- 迁移文件 `0003_fluffy_wildside.sql` 包含添加 `password` 列的语句
- 但之前运行 `db:migrate` 时失败（因为有重复的邮箱数据）

**解决方案**：
创建 `backend/fix-schema.js` 脚本手动修复：

```javascript
import postgres from 'postgres'
import dotenv from 'dotenv'

dotenv.config({ path: '.env' })
const sql = postgres(process.env.DATABASE_URL)

async function fixSchema() {
  // 1. 添加 password 列
  await sql`
    ALTER TABLE users
    ADD COLUMN IF NOT EXISTS password varchar(255)
  `

  // 2. 清理重复的邮箱数据
  await sql`
    DELETE FROM users a USING users b
    WHERE a.id > b.id AND a.email = b.email AND a.email IS NOT NULL
  `

  // 3. 添加 email 唯一约束
  await sql`
    ALTER TABLE users
    ADD CONSTRAINT users_email_unique UNIQUE(email)
  `
}
```

**执行结果**：
```bash
$ bun run fix-schema.js
🔧 开始修复数据库 schema...
✅ password 列已添加
⚠️  email 列有重复数据，清理中...
✅ 重复数据已清理
✅ email 唯一约束已添加
🎉 数据库 schema 修复完成！
```
